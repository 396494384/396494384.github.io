>代码演式

 ```javascript
 //定义一个render函数
 function render(strTemplate, obj){
	var reg = /\{\{(\w+)\}\}/;
	while(reg.test(strTemplate)){
		var key = reg.exec(strTemplate)[1];
		strTemplate = strTemplate.replace(reg, obj[key]);
	}
	return strTemplate;
}

let template = '我是{{name}}，年龄{{age}}，性别{{sex}}';
let person = {
	name: '布兰',
	age: 12,
	sex:"男"
}
//调用render
document.getElementById("text").innerHTML = render(template, person); // 我是布兰，年龄12，性别男

```
